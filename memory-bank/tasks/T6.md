## Data Structure
```typescript
interface Bookmark {
  paperId: string;
  dateAdded: Date;
  title: string;    // For quick listing without loading paper
  category: string; // For filtering
  collections?: string[];
  paperData: any;   // Store the complete paper data
}

interface BookmarkStore {
  bookmarks: Record<string, Bookmark>;
  collections: string[];
}
```
